// ✅ Modelo de dados Remetente
const Remetente = mongoose.model('Remetente', {
  nome: {
    type: String,
    required: true
  },
  cnpj: {
    type: String,
    required: true
  },
  endereco: {
    type: String,
    required: true
  },
  municipio: {
    type: String,
    required: true
  },
  uf: {
    type: String,
    required: true,
    maxlength: 2
  }
});
